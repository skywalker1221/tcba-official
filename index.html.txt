    <script>
        // DOM 元素
        const videoUpload = document.getElementById('video-upload');
        const videoElement = document.getElementById('source-video');
        const canvas = document.getElementById('render-canvas');
        const ctx = canvas.getContext('2d');
        const playPauseBtn = document.getElementById('play-pause-btn');
        const playIcon = document.getElementById('play-icon');
        const exportBtn = document.getElementById('export-btn');
        const exportText = document.getElementById('export-text');
        const statusMsg = document.getElementById('status-msg');

        // 控制參數
        const scaleInput = document.getElementById('video-scale');
        const offsetXInput = document.getElementById('video-offset-x');
        const offsetYInput = document.getElementById('video-offset-y');
        
        const scaleVal = document.getElementById('scale-val');
        const offsetXVal = document.getElementById('offset-x-val');
        const offsetYVal = document.getElementById('offset-y-val');

        const fontSelect = document.getElementById('font-family-select');
        const topFillStyleSelect = document.getElementById('top-fill-style');
        const solidColorControls = document.getElementById('solid-color-controls');

        const topTextInput = document.getElementById('top-text');
        const topTextColor = document.getElementById('top-text-color');
        const topBorderColor = document.getElementById('top-border-color');
        const topYInput = document.getElementById('top-y');
        const topYVal = document.getElementById('top-y-val');

        const bottomTextInput = document.getElementById('bottom-text');
        const bottomTextColor = document.getElementById('bottom-text-color');
        const bottomBorderColor = document.getElementById('bottom-border-color');
        const bottomBgColor = document.getElementById('bottom-bg-color');
        const bottomYInput = document.getElementById('bottom-y');
        const bottomYVal = document.getElementById('bottom-y-val');

        const toggleSafeZone = document.getElementById('toggle-safe-zone');
        const ratioBtns = document.querySelectorAll('.ratio-btn');

        // 核心狀態
        let animationFrameId;
        let isRecording = false;
        let mediaRecorder;
        let recordedChunks = [];
        let videoLoaded = false;
        let recordingTimer = null;

        // 預設比例 4:5
        let currentBoxWidth = 720;
        let currentBoxHeight = 900; 

        // 初始化
        window.addEventListener('load', () => {
            drawFrame();
        });

        topFillStyleSelect.addEventListener('change', () => {
            if (topFillStyleSelect.value === 'solid') {
                solidColorControls.classList.remove('hidden');
            } else {
                solidColorControls.classList.add('hidden');
            }
            triggerRedraw();
        });

        fontSelect.addEventListener('change', () => {
            triggerRedraw();
        });

        document.fonts.ready.then(() => {
            drawFrame();
        });

        // 綁定手動滑桿
        const controls = [
            { el: scaleInput, valEl: scaleVal, suffix: 'x' },
            { el: offsetXInput, valEl: offsetXVal, suffix: 'px' },
            { el: offsetYInput, valEl: offsetYVal, suffix: 'px' },
            { el: topYInput, valEl: topYVal, prefix: 'Y: ' },
            { el: bottomYInput, valEl: bottomYVal, prefix: 'Y: ' }
        ];

        controls.forEach(ctrl => {
            ctrl.el.addEventListener('input', (e) => {
                let v = e.target.value;
                if (ctrl.suffix) v += ctrl.suffix;
                if (ctrl.prefix) v = ctrl.prefix + v;
                ctrl.valEl.textContent = v;
                triggerRedraw();
            });
        });

        [
            topTextInput, topTextColor, topBorderColor,
            bottomTextInput, bottomTextColor, bottomBorderColor, bottomBgColor,
            toggleSafeZone
        ].forEach(input => {
            input.addEventListener('input', triggerRedraw);
        });

        function triggerRedraw() {
            drawFrame();
        }

        // 一鍵固定尺寸
        ratioBtns.forEach(btn => {
            btn.addEventListener('click', () => {
                ratioBtns.forEach(b => {
                    b.classList.remove('border-blue-500', 'bg-blue-600/10', 'text-blue-400');
                    b.classList.add('border-gray-700', 'bg-gray-950/60', 'text-gray-300');
                });
                btn.classList.remove('border-gray-700', 'bg-gray-950/60', 'text-gray-300');
                btn.classList.add('border-blue-500', 'bg-blue-600/10', 'text-blue-400');

                const ratioStr = btn.getAttribute('data-ratio');
                const [wPart, hPart] = ratioStr.split('/').map(Number);
                const targetRatio = wPart / hPart;

                currentBoxWidth = 720;
                currentBoxHeight = Math.round(720 / targetRatio);

                if (currentBoxHeight > 1280) {
                    currentBoxHeight = 1280;
                    currentBoxWidth = Math.round(1280 * targetRatio);
                }

                scaleInput.value = "1.0";
                scaleVal.textContent = "1.0x";
                offsetXInput.value = "0";
                offsetXVal.textContent = "0px";
                offsetYInput.value = "0";
                offsetYVal.textContent = "0px";

                drawFrame();
            });
        });

        // 處理影片載入
        videoUpload.addEventListener('change', function(e) {
            const file = e.target.files[0];
            if (!file) return;

            videoLoaded = false;
            exportBtn.disabled = true;
            statusMsg.textContent = '解碼影片中，請稍候...';

            const url = URL.createObjectURL(file);
            videoElement.src = url;
            videoElement.load();
        });

        videoElement.addEventListener('loadedmetadata', () => {
            videoLoaded = true;
            exportBtn.disabled = false;
            playPauseBtn.classList.remove('hidden');
            statusMsg.textContent = '解碼成功！雙層描邊技術已啟動，完美相容 iOS 下載！';
            videoElement.currentTime = 0.1; 
        });

        videoElement.addEventListener('seeked', () => { drawFrame(); });
        videoElement.addEventListener('canplay', () => { drawFrame(); });
        videoElement.addEventListener('timeupdate', () => {
            if (videoElement.paused && !isRecording) {
                drawFrame();
            }
        });

        videoElement.addEventListener('play', () => {
            playIcon.innerHTML = '<path d="M6 19h4V5H6v14zm8-14v14h4V5h-4z"/>';
            renderLoop();
        });

        videoElement.addEventListener('pause', () => {
            playIcon.innerHTML = '<path d="M8 5v14l11-7z"/>';
            cancelAnimationFrame(animationFrameId);
            drawFrame();
        });

        playPauseBtn.addEventListener('click', () => {
            if (videoElement.paused) {
                videoElement.play();
            } else {
                videoElement.pause();
            }
        });

        function renderLoop() {
            drawFrame();
            if (!videoElement.paused && !videoElement.ended) {
                animationFrameId = requestAnimationFrame(renderLoop);
            }
        }

        // 核心渲染 Canvas 功能
        function drawFrame() {
            ctx.fillStyle = '#000000';
            ctx.fillRect(0, 0, canvas.width, canvas.height);

            const frameW = currentBoxWidth;
            const frameH = currentBoxHeight;
            const frameX = (canvas.width - frameW) / 2;
            const frameY = (canvas.height - frameH) / 2;

            // 1. 繪製影片區
            if (videoLoaded && videoElement.readyState >= 1) {
                ctx.save();
                ctx.beginPath();
                ctx.rect(frameX, frameY, frameW, frameH);
                ctx.clip();

                const vWidth = videoElement.videoWidth || 720;
                const vHeight = videoElement.videoHeight || 1280;
                const vRatio = vWidth / vHeight;
                const fRatio = frameW / frameH;
                let baseW, baseH;

                if (vRatio > fRatio) {
                    baseH = frameH;
                    baseW = frameH * vRatio;
                } else {
                    baseW = frameW;
                    baseH = frameW / vRatio;
                }

                const scale = parseFloat(scaleInput.value);
                const offX = parseFloat(offsetXInput.value);
                const offY = parseFloat(offsetYInput.value);

                const drawW = baseW * scale;
                const drawH = baseH * scale;
                const drawX = frameX + (frameW - drawW) / 2 + offX;
                const drawY = frameY + (frameH - drawH) / 2 + offY;

                try {
                    ctx.drawImage(videoElement, drawX, drawY, drawW, drawH);
                } catch (err) {
                    console.warn("無法繪製影格:", err);
                }
                ctx.restore();
            } else {
                ctx.save();
                ctx.strokeStyle = 'rgba(255, 255, 255, 0.25)';
                ctx.lineWidth = 3;
                ctx.setLineDash([15, 10]);
                ctx.strokeRect(frameX + 10, frameY + 10, frameW - 20, frameH - 20);
                
                ctx.fillStyle = 'rgba(255, 255, 255, 0.4)';
                ctx.font = 'bold 24px sans-serif';
                ctx.textAlign = 'center';
                ctx.fillText('【 影片顯示預留區（置中 4:5 ） 】', canvas.width / 2, canvas.height / 2);
                ctx.restore();
            }

            // 取得選擇的字型設置
            const chosenFont = fontSelect.value;

            // ==========================================
            // 2. 繪製上方日式大標題 (100% 解決 iOS 變黑 Bug)
            // ==========================================
            const topText = topTextInput.value;
            if (topText) {
                const topLines = topText.split('\n');
                const numLines = topLines.length;

                // 根據行數動態縮放字體與行高，確保多行字完美塞在標題區
                let fontSize = 76;
                let lineHeight = 95;
                if (numLines === 2) {
                    fontSize = 54;    
                    lineHeight = 68;  
                } else if (numLines >= 3) {
                    fontSize = 38;
                    lineHeight = 48;
                }

                ctx.textAlign = 'center';
                ctx.textBaseline = 'middle';
                ctx.font = `900 ${fontSize}px ${chosenFont}`;

                // 以「整組文字的垂直中心點 Y」為基準
                const centerY = parseInt(topYInput.value); 
                const totalTextHeight = (numLines - 1) * lineHeight;
                const startY = centerY - (totalTextHeight / 2);

                topLines.forEach((line, index) => {
                    const textY = startY + (index * lineHeight);
                    
                    // 【終極解決方案】：不用 shadowBlur，改用「多層透明漸層線條疊加」來做白色外發光！
                    // 這在 iOS Safari、Chrome、甚至舊版手機瀏覽器中 100% 絕對穩定、不黑屏不切邊！
                    ctx.save();
                    ctx.lineJoin = 'round';
                    ctx.miterLimit = 2;
                    
                    // 第一層（最淡、最寬的外發光圈）
                    ctx.strokeStyle = 'rgba(255, 255, 255, 0.15)';
                    ctx.lineWidth = fontSize * 0.45;
                    ctx.strokeText(line, canvas.width / 2, textY);
                    
                    // 第二層（中等強度外光圈）
                    ctx.strokeStyle = 'rgba(255, 255, 255, 0.35)';
                    ctx.lineWidth = fontSize * 0.35;
                    ctx.strokeText(line, canvas.width / 2, textY);
                    
                    // 第三層（核心實體白邊）
                    ctx.strokeStyle = 'rgba(255, 255, 255, 0.7)';
                    ctx.lineWidth = fontSize * 0.25;
                    ctx.strokeText(line, canvas.width / 2, textY);
                    ctx.restore();
                    
                    // B. 中間層：純黑銳利外框 (Sharp Black Outline)
                    ctx.save();
                    ctx.lineJoin = 'round';
                    ctx.miterLimit = 2;
                    ctx.lineWidth = fontSize * 0.18;
                    ctx.strokeStyle = (topFillStyleSelect.value === 'solid') ? topBorderColor.value : '#000000';
                    ctx.strokeText(line, canvas.width / 2, textY);
                    ctx.restore();
                    
                    // C. 最表層：字體填充 (銀色漸層/金色漸層/純色)
                    ctx.save();
                    if (topFillStyleSelect.value === 'silver') {
                        // 精密亮色金屬銀色漸層 (純白頂部 -> 亮銀白 -> 銀灰)
                        const grad = ctx.createLinearGradient(0, textY - (fontSize/2), 0, textY + (fontSize/2));
                        grad.addColorStop(0, '#ffffff');     // 純白
                        grad.addColorStop(0.3, '#f5f5f7');   // 極淺亮銀
                        grad.addColorStop(0.65, '#d0d0d8');  // 亮銀
                        grad.addColorStop(1, '#a0a0af');      // 經典金屬銀
                        ctx.fillStyle = grad;
                    } else if (topFillStyleSelect.value === 'gold') {
                        // 奢華金色漸層
                        const grad = ctx.createLinearGradient(0, textY - (fontSize/2), 0, textY + (fontSize/2));
                        grad.addColorStop(0, '#fff6d1');    
                        grad.addColorStop(0.4, '#ffd700');  
                        grad.addColorStop(0.7, '#e6b800');  
                        grad.addColorStop(1, '#9e7800');     
                        ctx.fillStyle = grad;
                    } else {
                        ctx.fillStyle = topTextColor.value;
                    }
                    ctx.fillText(line, canvas.width / 2, textY);
                    ctx.restore();
                });
            }

            // ==========================================
            // 3. 繪製下方日式標籤 (完美鎖定「思源宋體」，絕無線條交叉混亂)
            // ==========================================
            const bottomText = bottomTextInput.value;
            if (bottomText) {
                // 字型一律鎖定為極高雅、清晰的思源宋體 Noto Serif TC
                const bottomFont = "'Noto Serif TC', serif";
                ctx.font = `900 75px ${bottomFont}`;
                ctx.textAlign = 'center';
                ctx.textBaseline = 'middle';

                const textMetrics = ctx.measureText(bottomText);
                const paddingX = 65;
                const paddingY = 25;
                const boxW = textMetrics.width + paddingX * 2;
                const boxH = 120;
                const boxX = (canvas.width - boxW) / 2;
                const boxY = parseInt(bottomYInput.value) - boxH / 2;

                // A. 圓角底板外黑影
                ctx.save();
                ctx.shadowColor = '#000000';
                ctx.shadowBlur = 25;
                ctx.fillStyle = bottomBgColor.value;
                ctx.beginPath();
                ctx.roundRect(boxX, boxY, boxW, boxH, 20);
                ctx.fill();
                ctx.restore();

                // B. 底板外細黑框
                ctx.lineWidth = 4;
                ctx.strokeStyle = '#000000';
                ctx.stroke();

                const textY = parseInt(bottomYInput.value);

                // C. 繪製文字粗外框 (描邊底層 - 白框)
                // 先畫白色外框，將 lineJoin 設為圓角
                ctx.save();
                ctx.lineJoin = 'round';
                ctx.miterLimit = 2;
                ctx.lineWidth = 15; // 完美的粗細設定，不會粗到讓筆劃打結
                ctx.strokeStyle = bottomBorderColor.value;
                ctx.strokeText(bottomText, canvas.width / 2, textY);
                ctx.restore();

                // D. 疊加填滿主體字 (填滿表層 - 紅字)
                // 由於是先畫「寬框」，隨後在上方「填滿」覆蓋，字體內部的黑色交叉線會被 100% 蓋掉，呈現絕對乾淨、高級的邊框字！
                ctx.save();
                ctx.fillStyle = bottomTextColor.value;
                ctx.fillText(bottomText, canvas.width / 2, textY);
                ctx.restore();
                
                // E. 精細漫畫黑描邊 (最表層細黑邊框，增加字體立體層次，還原照片細節)
                ctx.save();
                ctx.lineJoin = 'round';
                ctx.lineWidth = 2;
                ctx.strokeStyle = '#000000';
                ctx.strokeText(bottomText, canvas.width / 2, textY);
                ctx.restore();
            }

            // 4. 繪製 IG Reels 輔助安全線
            if (toggleSafeZone.checked && !isRecording) {
                ctx.save();
                
                ctx.fillStyle = 'rgba(239, 68, 68, 0.15)';
                ctx.fillRect(0, 0, canvas.width, 140);
                ctx.strokeStyle = 'rgba(239, 68, 68, 0.4)';
                ctx.lineWidth = 2;
                ctx.beginPath();
                ctx.moveTo(0, 140); ctx.lineTo(canvas.width, 140);
                ctx.stroke();
                
                const safeBottomY = canvas.height - 280;
                ctx.fillStyle = 'rgba(239, 68, 68, 0.15)';
                ctx.fillRect(0, safeBottomY, canvas.width, 280);
                ctx.beginPath();
                ctx.moveTo(0, safeBottomY); ctx.lineTo(canvas.width, safeBottomY);
                ctx.stroke();

                ctx.fillRect(canvas.width - 130, 140, 130, safeBottomY - 140);
                ctx.beginPath();
                ctx.moveTo(canvas.width - 130, 140); ctx.lineTo(canvas.width - 130, safeBottomY);
                ctx.stroke();

                ctx.fillStyle = '#f87171';
                ctx.font = 'bold 22px sans-serif';
                ctx.textAlign = 'left';
                ctx.fillText('⚠️ IG 頂部 UI 遮擋區', 20, 45);
                ctx.fillText('⚠️ 右側按鈕區', canvas.width - 200, 350);
                ctx.fillText('⚠️ IG 底部頭貼/文字區', 20, canvas.height - 40);

                ctx.restore();
            }
        }

        // ==========================================
        // 穩定導出影片邏輯
        // ==========================================
        exportBtn.addEventListener('click', () => {
            if (!videoLoaded) return;
            
            if (isRecording) {
                stopRecording();
            } else {
                startRecording();
            }
        });

        function startRecording() {
            isRecording = true;
            
            exportBtn.classList.remove('from-blue-600', 'to-purple-600');
            exportBtn.classList.add('from-red-600', 'to-orange-600');
            exportText.textContent = "錄製中，點擊可強制導出影片";
            statusMsg.textContent = "正在將完美置中與白色外發光漸層字體壓製成短影音中...";
            playPauseBtn.classList.add('hidden');

            videoElement.pause();
            videoElement.currentTime = 0;
            
            videoElement.addEventListener('seeked', function startActualRecord() {
                videoElement.removeEventListener('seeked', startActualRecord);
                
                drawFrame();

                const fps = 30;
                const canvasStream = canvas.captureStream(fps);
                let combinedStream;

                try {
                    let videoStream = null;
                    if (videoElement.captureStream) {
                        videoStream = videoElement.captureStream();
                    } else if (videoElement.mozCaptureStream) {
                        videoStream = videoElement.mozCaptureStream();
                    }

                    if (videoStream) {
                        const audioTracks = videoStream.getAudioTracks();
                        if (audioTracks.length > 0) {
                            combinedStream = new MediaStream([
                                ...canvasStream.getVideoTracks(),
                                ...audioTracks
                            ]);
                        } else {
                            combinedStream = canvasStream;
                        }
                    } else {
                        combinedStream = canvasStream;
                    }
                } catch (err) {
                    combinedStream = canvasStream;
                }

                let options = { mimeType: 'video/webm; codecs=vp9' };
                if (MediaRecorder.isTypeSupported('video/mp4; codecs=avc1.42E01E,mp4a.40.2')) {
                    options = { mimeType: 'video/mp4; codecs=avc1.42E01E,mp4a.40.2' };
                } else if (MediaRecorder.isTypeSupported('video/mp4')) {
                    options = { mimeType: 'video/mp4' };
                } else if (MediaRecorder.isTypeSupported('video/webm; codecs=h264')) {
                    options = { mimeType: 'video/webm; codecs=h264' };
                }

                recordedChunks = [];
                try {
                    mediaRecorder = new MediaRecorder(combinedStream, options);
                } catch (e) {
                    mediaRecorder = new MediaRecorder(combinedStream);
                }

                mediaRecorder.ondataavailable = (event) => {
                    if (event.data && event.data.size > 0) {
                        recordedChunks.push(event.data);
                    }
                };

                mediaRecorder.onstop = () => {
                    downloadVideo();
                };

                mediaRecorder.start();
                videoElement.play();

                videoElement.addEventListener('timeupdate', checkRecordingEnd);

                const durationMs = (videoElement.duration || 10) * 1000;
                recordingTimer = setTimeout(() => {
                    stopRecording();
                }, durationMs + 2000);
            });
        }

        function checkRecordingEnd() {
            if (videoElement.currentTime >= videoElement.duration - 0.1 || videoElement.ended) {
                stopRecording();
            }
        }

        function stopRecording() {
            if (recordingTimer) {
                clearTimeout(recordingTimer);
                recordingTimer = null;
            }

            videoElement.removeEventListener('timeupdate', checkRecordingEnd);
            
            if (isRecording) {
                isRecording = false;
                if (mediaRecorder && mediaRecorder.state !== 'inactive') {
                    mediaRecorder.stop();
                }
                videoElement.pause();
                playPauseBtn.classList.remove('hidden');
            }
        }

        function downloadVideo() {
            const mimeType = mediaRecorder.mimeType || 'video/mp4';
            const isMp4 = mimeType.includes('mp4');
            const ext = isMp4 ? 'mp4' : 'webm';

            if (recordedChunks.length === 0) {
                statusMsg.textContent = "❌ 錄製失敗，請重新載入影片重試。";
                resetButtonState();
                return;
            }

            const blob = new Blob(recordedChunks, { type: mimeType });
            const url = URL.createObjectURL(blob);
            
            const a = document.createElement('a');
            document.body.appendChild(a);
            a.style = 'display: none';
            a.href = url;
            a.download = `IG-Reels-Japanese-${Date.now()}.${ext}`;
            a.click();
            
            setTimeout(() => {
                window.URL.revokeObjectURL(url);
                document.body.removeChild(a);
            }, 100);

            resetButtonState();
            statusMsg.textContent = `🎉 成功導出！已儲存為高畫質日系綜藝短影音 (${ext.toUpperCase()})。`;
            drawFrame();
        }

        function resetButtonState() {
            exportBtn.classList.remove('from-red-600', 'to-orange-600');
            exportBtn.classList.add('from-blue-600', 'to-purple-600');
            exportText.textContent = "匯出並下載完美 MP4";
            exportBtn.disabled = false;
        }
    </script>
